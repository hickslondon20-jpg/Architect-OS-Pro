## Goal

Create the `skill-files` Storage bucket and its `skill_files` metadata table for a skill's own
building-block resources (scripts/references/assets) — structurally separate from KB documents, per
the project CONTEXT.md's hard line between "founder business data" (KB) and "a skill's own
resources" (skill files). Implement the one genuinely new RLS shape this build needs: global skill
files are open-read, admin-only-write.

## Migration

File: `docs/migrations/20260701_skill_files_storage.sql`

```sql
-- Phase 1 (Agent Skills & Document Generation Engine): skill-files bucket + skill_files metadata.

-- 1. Bucket (private — access controlled entirely by RLS below, not bucket-level public flag)
INSERT INTO storage.buckets (id, name, public)
VALUES ('skill-files', 'skill-files', false)
ON CONFLICT (id) DO NOTHING;

-- 2. Metadata table
CREATE TABLE IF NOT EXISTS public.skill_files (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  skill_id      UUID NOT NULL REFERENCES public.skill_packs(id) ON DELETE CASCADE,
  filename      TEXT NOT NULL,
  category      TEXT NOT NULL CHECK (category IN ('scripts', 'references', 'assets')),
  mime_type     TEXT,
  size          BIGINT,
  storage_path  TEXT NOT NULL,             -- {owner_user_id}/{skill_id}/{category}/{filename}
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS skill_files_skill_id_idx ON public.skill_files (skill_id);
CREATE INDEX IF NOT EXISTS skill_files_category_idx ON public.skill_files (category);

CREATE OR REPLACE FUNCTION public.update_skill_files_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER skill_files_updated_at_trigger
  BEFORE UPDATE ON public.skill_files
  FOR EACH ROW
  EXECUTE FUNCTION public.update_skill_files_updated_at();

ALTER TABLE public.skill_files ENABLE ROW LEVEL SECURITY;

-- 3. RLS on skill_files (metadata table) — mirrors skill_packs' own scope/ownership split
CREATE POLICY "skill_files_select_own_or_global"
  ON public.skill_files FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM public.skill_packs sp
      WHERE sp.id = skill_files.skill_id
        AND (sp.scope = 'global' OR sp.user_id = (SELECT auth.uid()))
    )
  );

CREATE POLICY "skill_files_insert_own"
  ON public.skill_files FOR INSERT
  WITH CHECK (
    EXISTS (
      SELECT 1 FROM public.skill_packs sp
      WHERE sp.id = skill_files.skill_id AND sp.user_id = (SELECT auth.uid())
    )
  );

CREATE POLICY "skill_files_update_own"
  ON public.skill_files FOR UPDATE
  USING (
    EXISTS (SELECT 1 FROM public.skill_packs sp WHERE sp.id = skill_files.skill_id AND sp.user_id = (SELECT auth.uid()))
  )
  WITH CHECK (
    EXISTS (SELECT 1 FROM public.skill_packs sp WHERE sp.id = skill_files.skill_id AND sp.user_id = (SELECT auth.uid()))
  );

CREATE POLICY "skill_files_delete_own"
  ON public.skill_files FOR DELETE
  USING (
    EXISTS (SELECT 1 FROM public.skill_packs sp WHERE sp.id = skill_files.skill_id AND sp.user_id = (SELECT auth.uid()))
  );

-- 4. RLS on storage.objects for the skill-files bucket
-- Private-skill-file access: owner-only read+write, mirroring kb_files_*_own_folder exactly.
CREATE POLICY "skill_files_select_own_folder"
  ON storage.objects FOR SELECT
  USING (bucket_id = 'skill-files' AND (storage.foldername(name))[1] = (SELECT auth.uid())::text);

CREATE POLICY "skill_files_insert_own_folder"
  ON storage.objects FOR INSERT
  WITH CHECK (bucket_id = 'skill-files' AND (storage.foldername(name))[1] = (SELECT auth.uid())::text);

CREATE POLICY "skill_files_update_own_folder"
  ON storage.objects FOR UPDATE
  USING (bucket_id = 'skill-files' AND (storage.foldername(name))[1] = (SELECT auth.uid())::text)
  WITH CHECK (bucket_id = 'skill-files' AND (storage.foldername(name))[1] = (SELECT auth.uid())::text);

CREATE POLICY "skill_files_delete_own_folder"
  ON storage.objects FOR DELETE
  USING (bucket_id = 'skill-files' AND (storage.foldername(name))[1] = (SELECT auth.uid())::text);

-- Global-skill-file access (the genuinely new shape — open-read, admin-only-write):
CREATE POLICY "skill_files_select_global_folder"
  ON storage.objects FOR SELECT
  USING (
    bucket_id = 'skill-files' AND EXISTS (
      SELECT 1 FROM public.profiles p
      WHERE p.user_id::text = (storage.foldername(name))[1] AND p.is_admin = true
    )
  );
-- Note: the admin's own folder is covered by BOTH this policy and the owner-only policy above
-- (Postgres RLS policies are OR'd together per command) — this is intentional, not a conflict:
-- the admin reads/writes their own folder via the owner policy; every OTHER founder reads it via
-- this global policy. No separate admin-write policy needed beyond the owner-only INSERT/UPDATE/
-- DELETE policies above, since the admin IS the folder owner for their own global skill files.

-- Rollback:

-- DROP POLICY IF EXISTS "skill_files_insert_own_folder" ON storage.objects;
-- DROP POLICY IF EXISTS "skill_files_update_own_folder" ON storage.objects;
-- DROP POLICY IF EXISTS "skill_files_delete_own_folder" ON storage.objects;
```

## Success Criteria

1. `skill-files` bucket exists, private.
2. `skill_files` table exists with the specified columns, FK to `skill_packs`, RLS enabled.
3. A founder JWT can upload a file to their own `{their_user_id}/{their_skill_id}/scripts/...` path
   and read it back.
4. A founder JWT can read a file under the admin's `{admin_user_id}/{admin_skill_id}/...` path
   (global skill file) but **cannot** write/delete it.
5. A founder JWT cannot read another non-admin founder's private skill file path.
6. `skill_files` metadata rows follow the same scope/ownership visibility as their parent
   `skill_packs` row (global skill's files' metadata readable by all; private skill's files' metadata
   readable only by its owner).
7. `generate_typescript_types` re-run; `skill_files` reflected in generated types.
