## Data Models (internal to the service)

```python
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Literal, Any

@dataclass(frozen=True)
class KbFolderItem:
    type: Literal["folder"]
    id: str
    name: str

@dataclass(frozen=True)
class KbFileItem:
    type: Literal["file"]
    id: str
    name: str
    file_type: str
    status: str

KbItem = KbFolderItem | KbFileItem

@dataclass(frozen=True)
class KbLsResult:
    folder_id: str | None
    folder_name: str
    item_count: int
    items: list[KbItem]

@dataclass
class KbTreeNode:
    type: Literal["folder", "file"]
    id: str
    name: str
    file_type: str | None = None
    status: str | None = None
    children: list["KbTreeNode"] = field(default_factory=list)

@dataclass(frozen=True)
class KbTreeResult:
    folder_id: str | None
    folder_name: str
    depth_requested: int
    total_items: int
    truncated: bool
    tree: list[KbTreeNode]
```

## `KbNavigationService` Class

```python
class KbNavigationService:
    def __init__(self, store: VectorStore) -> None:
        self.client = store.client

    @classmethod
    def from_env(cls) -> "KbNavigationService":
        from services.vector_store import VectorStore
        return cls(VectorStore.from_env())

    def execute_ls(self, user_id: str, folder_id: str | None) -> KbLsResult:
        """List immediate children of a folder. folder_id=None = root."""
        folder_name = self._resolve_folder_name(user_id, folder_id)

        # 1. Child folders
        folder_query = (
            self.client.table("kb_folders")
            .select("id,name")
            .eq("user_id", user_id)
            .order("name")
        )
        if folder_id is None:
            folder_query = folder_query.is_("parent_id", "null")
        else:
            folder_query = folder_query.eq("parent_id", folder_id)
        folders_resp = folder_query.execute()

        # 2. Files in this folder
        file_query = (
            self.client.table("ose_raw_document_registry")
            .select("id,file_name,file_type,status")
            .eq("user_id", user_id)
            .neq("status", "deleted")
            .order("file_name")
        )
        if folder_id is None:
            file_query = file_query.is_("folder_id", "null")
        else:
            file_query = file_query.eq("folder_id", folder_id)
        files_resp = file_query.execute()

        items: list[KbItem] = []
        for row in folders_resp.data or []:
            items.append(KbFolderItem(type="folder", id=row["id"], name=row["name"]))
        for row in files_resp.data or []:
            items.append(KbFileItem(
                type="file",
                id=row["id"],
                name=row["file_name"],
                file_type=row.get("file_type") or "",
                status=row.get("status") or "",
            ))

        return KbLsResult(
            folder_id=folder_id,
            folder_name=folder_name,
            item_count=len(items),
            items=items,
        )

    def execute_tree(
        self,
        user_id: str,
        folder_id: str | None,
        depth: int = 3,
        limit: int = 200,
    ) -> KbTreeResult:
        """Return nested tree from folder_id down to depth levels."""
        depth = max(1, min(depth, 10))
        limit = max(1, min(limit, 500))

        folder_name = self._resolve_folder_name(user_id, folder_id)

        # Fetch ALL folders and files for this user in one round-trip each.
        # Build the tree in Python — avoids N+1 queries.
        all_folders_resp = (
            self.client.table("kb_folders")
            .select("id,name,parent_id")
            .eq("user_id", user_id)
            .execute()
        )
        all_files_resp = (
            self.client.table("ose_raw_document_registry")
            .select("id,file_name,file_type,status,folder_id")
            .eq("user_id", user_id)
            .neq("status", "deleted")
            .execute()
        )

        folders_by_parent: dict[str | None, list[dict]] = {}
        for row in all_folders_resp.data or []:
            key = row.get("parent_id")  # None for root-level
            folders_by_parent.setdefault(key, []).append(row)

        files_by_folder: dict[str | None, list[dict]] = {}
        for row in all_files_resp.data or []:
            key = row.get("folder_id")  # None for root-level
            files_by_folder.setdefault(key, []).append(row)

        item_counter = [0]
        truncated = [False]

        def _build(current_folder_id: str | None, current_depth: int) -> list[KbTreeNode]:
            if current_depth > depth or truncated[0]:
                return []
            nodes: list[KbTreeNode] = []

            # Child folders first, alphabetical
            child_folders = sorted(
                folders_by_parent.get(current_folder_id, []),
                key=lambda r: r["name"].lower(),
            )
            for row in child_folders:
                if truncated[0]:
                    break
                item_counter[0] += 1
                if item_counter[0] > limit:
                    truncated[0] = True
                    break
                node = KbTreeNode(type="folder", id=row["id"], name=row["name"])
                node.children = _build(row["id"], current_depth + 1)
                nodes.append(node)

            # Then files, alphabetical
            child_files = sorted(
                files_by_folder.get(current_folder_id, []),
                key=lambda r: r["file_name"].lower(),
            )
            for row in child_files:
                if truncated[0]:
                    break
                item_counter[0] += 1
                if item_counter[0] > limit:
                    truncated[0] = True
                    break
                nodes.append(KbTreeNode(
                    type="file",
                    id=row["id"],
                    name=row["file_name"],
                    file_type=row.get("file_type") or "",
                    status=row.get("status") or "",
                ))
            return nodes

        tree_nodes = _build(folder_id, 1)
        return KbTreeResult(
            folder_id=folder_id,
            folder_name=folder_name,
            depth_requested=depth,
            total_items=item_counter[0],
            truncated=truncated[0],
            tree=tree_nodes,
        )

    def _resolve_folder_name(self, user_id: str, folder_id: str | None) -> str:
        if folder_id is None:
            return "Knowledge Base"
        resp = (
            self.client.table("kb_folders")
            .select("name")
            .eq("id", folder_id)
            .eq("user_id", user_id)
            .limit(1)
            .execute()
        )
        rows = resp.data or []
        if not rows:
            raise KbNavigationError(f"Folder {folder_id!r} not found for this user.")
        return rows[0]["name"]


class KbNavigationError(RuntimeError):
    pass
```

## Serialization Helpers

Add to `folder_navigation.py` — used by the FastAPI endpoints to convert dataclasses to JSON-serializable dicts:

```python
def _item_to_dict(item: KbItem) -> dict:
    if item.type == "folder":
        return {"type": "folder", "id": item.id, "name": item.name}
    return {"type": "file", "id": item.id, "name": item.name, "file_type": item.file_type, "status": item.status}

def _tree_node_to_dict(node: KbTreeNode) -> dict:
    d: dict = {"type": node.type, "id": node.id, "name": node.name}
    if node.file_type is not None:
        d["file_type"] = node.file_type
    if node.status is not None:
        d["status"] = node.status
    if node.children:
        d["children"] = [_tree_node_to_dict(child) for child in node.children]
    return d

def ls_result_to_dict(result: KbLsResult) -> dict:
    return {
        "folder_id": result.folder_id,
        "folder_name": result.folder_name,
        "item_count": result.item_count,
        "items": [_item_to_dict(item) for item in result.items],
    }

def tree_result_to_dict(result: KbTreeResult) -> dict:
    return {
        "folder_id": result.folder_id,
        "folder_name": result.folder_name,
        "depth_requested": result.depth_requested,
        "total_items": result.total_items,
        "truncated": result.truncated,
        "tree": [_tree_node_to_dict(node) for node in result.tree],
    }
```

## Verification Steps

1. **Compile check**: `python -m compileall python-backend/services/folder_navigation.py` — no syntax errors.
2. **`ls` at root (empty)**: Call `execute_ls(user_id, None)` — returns `KbLsResult` with `item_count=0`, `items=[]`, `folder_name="Knowledge Base"`.
3. **`ls` at root (with content)**: After creating test folders and uploading files (or inserting test rows), call `execute_ls` — folders appear before files, both alphabetical.
4. **`ls` on a specific folder**: Call with a real folder UUID — returns only that folder's immediate children.
5. **`ls` on nonexistent folder**: Not applicable (folders are not validated in `execute_ls` — if `folder_id` doesn't exist, returns empty result, which is correct).
6. **`tree` at root**: Call `execute_tree(user_id, None, depth=2)` — returns nested structure, folders before files at each level.
7. **`tree` depth limiting**: Create a 4-level folder nesting. Call `execute_tree(depth=2)` — items at level 3+ not included, `truncated` is false (depth is respected, not limit).
8. **`tree` limit truncation**: Call `execute_tree(limit=3)` with more than 3 total items — `truncated=True`, `total_items=3`.
9. **`_resolve_folder_name` wrong user**: Call with a `folder_id` belonging to a different user — raises `KbNavigationError`.
10. **No N+1 queries**: Confirm `execute_tree` makes exactly 3 Supabase calls (1 folder name resolve + 1 folders bulk fetch + 1 files bulk fetch) regardless of tree size.
